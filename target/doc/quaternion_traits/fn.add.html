Adds two quaternions.

use quaternion_traits::add;
 
let a: [f32; 4] = [1.0, 2.0, 3.0, 4.0];
let b: [f32; 4] = [4.0, 3.0, 2.0, -4.0];
let c: [f32; 4] = add::<f32, [f32; 4]>(&a, &b);
 
assert_eq!( c, [5.0, 5.0, 5.0, 0.0] );
